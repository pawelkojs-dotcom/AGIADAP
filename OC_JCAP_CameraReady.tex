\begin{abstract}
We present the Ontogenesis of Coherence (OC) framework, an adaptive scalar--tensor theory 
in which the ontogenic field $\sigma(x,t)$ encodes the dimensional coherence of spacetime 
and drives screening, late-time deviations, and observable signatures (CR1--CR3).
This manuscript is provided in camera-ready format for JCAP, complete with figures,
appendices, and all structural requirements.
\end{abstract}

\section*{1 \quad Introduction}
OC treats gravitational physics as an adaptation of geometric coherence regulated by $\sigma$.
Deviations arise in low-density regions through variations of the effective Planck mass $M_*^2(\sigma)$.

\section*{2 \quad Action}
\[
S = \int d^4x\sqrt{-g}\left[
\frac12 M_*^2(\sigma) R - \frac12(\nabla\sigma)^2 - V(\sigma)
- \frac12 c_T T^2 \sigma^2
\right] + S_m[g,\psi].
\]

\section*{3 \quad Field Equations}
\[
M_*^2 G_{\mu\nu} = T_{\mu\nu}^{(m)} + T_{\mu\nu}^{(\sigma)} 
- (\nabla_\mu\nabla_\nu - g_{\mu\nu}\Box)M_*^2,
\]
\[
\Box\sigma - V'(\sigma) - c_T T^2 \sigma + \frac12 R\frac{dM_*^2}{d\sigma} = 0.
\]

\section*{4 \quad Screening}
\[
m_{\rm eff}^2 = V'' + c_T T^2 + \frac12 R M_*^{2\,\prime\prime} + \frac{\beta^2 \rho}{M_{\rm Pl}^2}.
\]
Dense regions: $\sigma$ frozen. Voids: $\sigma$ plastic.

\section*{5 \quad FLRW Evolution}
\[
3H^2 = \frac{1}{M_*^2}(\rho_m+\rho_r+\rho_\sigma).
\]

\Phi = \Phi_N + \frac{\beta\delta\sigma}{M_{\rm Pl}}.
